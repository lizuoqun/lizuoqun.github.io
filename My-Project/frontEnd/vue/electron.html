添加配置 ​

添加启动命令， 在package.json中添加

  • main字段，指定主线程文件
  • scripts字段，指定启动命令 分别加上下面几个
  • build字段，指定打包配置
json
{
  "name": "web-vite-vue3",
  "version": "0.0.1",
  "main": "electron.js",
  "scripts": {
    "dev": "vite",
    "build": "vite build",
    "exe:start": "electron .",
    "exe:build": "electron-builder"
  },
  "dependencies": {},
  "devDependencies": {
    "electron": "^30.0.6",
    "electron-builder": "^24.13.3"
  },
  "build": {
    "productName": "760平台",
    "appId": "com.hailiao",
    "copyright": "by GuangDongHaiLiao",
    "win": {
      "icon": "public/gis-scenes.ico"
    },
    "nsis": {
      "oneClick": false,
      "allowElevation": true,
      "allowToChangeInstallationDirectory": true,
      "installerIcon": "public/gis-scenes.ico",
      "uninstallerIcon": "public/gis-scenes.ico",
      "installerHeaderIcon": "public/gis-scenes.ico"
    }
  }
}
