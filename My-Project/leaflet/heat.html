使用 ​

js
import 'leaflet.heat';

const heatPoint = [];
for (let i = 0; i < 1000; i++) {
  heatPoint.push([
    Math.random() * 80 - 40,
    Math.random() * 160 - 80,
    Math.random() * 3000 + ''
  ]);
}

function addHeat() {
  const layerHeat = L.heatLayer(heatPoint, {radius: 10});
  map.addLayer(layerHeat);
}
